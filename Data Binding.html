How Data-Binding is Implemented

AngularJS



$scope.C = 0;

$scope.F = function(){
    return $scope.C*9/5 + 32;
}

EmberJS



Ember.Object.create({
    C: 0,

    F: function(){
        return this.get('C')*9/5 
        + 32;
    }.property('C')
});
