Computed Properties in more Detail

AngularJS



$scope.C = 0;

$scope.F = function(){
    return $scope.C*9/5 + 32;
}

EmberJS



Ember.Object.create({
    C: 0,

    F: function(){
        return this.get('C')*9/5 
        + 32;
    }.property('C')
});

EmberJS



Ember.Object.create({

    if_A_then_B_else_C: function(){

        if(this.get('A')){
          return this.get('B');
        }else{
          return this.get('C');
        }

    }.property('A', 'B', 'C')
});

KnockoutJS



var C = ko.observable(0);

var F = ko.computed(function(){
    return C()*9/5 + 32;
});

KnockoutJSJS



var A = ko.observable(true);
var B = ko.observable(0);
var C = ko.observable(0);

var if_A_then_B_else_C = ko.computed(function(){

    if(A()){
      return B();
    }else{
      return C();
    }

});
